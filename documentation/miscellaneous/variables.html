src/common/decorators/route-doc.decorator.ts

RouteDoc
Default value : ({ operation, conflict, notFound, ok, created, accepted, }: IRoteDocOptions) => { const decorators = [] if (operation) decorators.push(ApiOperation({ description: operation })) if (conflict) decorators.push(ApiConflictResponse({ description: 'Already exists' })) if (notFound) decorators.push(ApiNotFoundResponse({ description: 'Not found' })) if (ok) { decorators.push(ApiOkResponse({ description: 'Ok', type: ok.type, isArray: ok.isArray })) } if (created) { decorators.push( ApiCreatedResponse({ description: 'Created', type: created.type, isArray: created.isArray }), ) } if (accepted) { decorators.push( ApiAcceptedResponse({ description: 'Accepted', type: accepted.type, isArray: accepted.isArray, }), ) } return applyDecorators(...decorators) }
